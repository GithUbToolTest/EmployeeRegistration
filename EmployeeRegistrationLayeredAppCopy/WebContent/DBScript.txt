CREATE TABLE "SYSTEM"."LAYERED_EMPLOYEE" 
   (	"EMPNO" NUMBER NOT NULL ENABLE, 
	"ENAME" VARCHAR2(20 BYTE), 
	"ADDRS" VARCHAR2(20 BYTE), 
	"DOJ" DATE, 
	"BASICS_SALARY" NUMBER, 
	"GROSS_SALARY" NUMBER, 
	"NET_SALARY" NUMBER, 
	 CONSTRAINT "LAYERED_EMPLOYEE_PK" PRIMARY KEY ("EMPNO")
CREATE SEQUENCE  "SYSTEM"."EMPNO_SEQ"  MINVALUE 1000 MAXVALUE 999999 INCREMENT BY 1 START WITH 1000 CACHE 20 NOORDER  NOCYCLE ;
